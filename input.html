<html>
    <script>
      function simpanData(event) {
        event.preventDefault(); // mencegah reload
        const npm = document.getElementById("npm").value;
        const nama = document.getElementById("nama").value;
        const alamat = document.getElementById("alamat").value;
        const notelp = document.getElementById("notelp").value;

        // Simpan ke localStorage (simpan sebagai array of object)
        const dataKaryawan = JSON.parse(localStorage.getItem("dataKaryawan")) || [];

        dataKaryawan.push({
          npm,
          nama,
          alamat,
          notelp
        });

        localStorage.setItem("dataKaryawan", JSON.stringify(dataKaryawan));

        alert("Data berhasil disimpan!");
        window.location.href = "input.html";
      }
    </script>
    <body>
