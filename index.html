    <script>
    // Inisialisasi data hanya sekali (jika belum ada)
    if (!localStorage.getItem("dataKaryawan")) {
      const initialData = [
        { npm: "12345678", nama: "LEO", alamat: "Amerika Serikat", notelp: "12345678765" },//data awal kalo misalnya belom input di input html
        { npm: "12345670", nama: "GUNA", alamat: "Amerika Serikat", notelp: "98765456789" }
      ];
      localStorage.setItem("dataKaryawan", JSON.stringify(initialData));
    }

    // Ambil ulang data setelah inisialisasi
    let data = JSON.parse(localStorage.getItem("dataKaryawan")) || [];

    // Tampilkan ke tabel
    const tabel = document.getElementById("tabelData");
    data.forEach((item, index) => {
      const row = document.createElement("tr");
      row.innerHTML = `
        <td>${item.npm}</td>
        <td>${item.nama}</td>
        <td>${item.alamat}</td>
        <td>${item.notelp}</td>
        <td>
          <button class="btn-edit" onclick="editData(${index})">EDIT</button>
          <button class="btn-delete" onclick="hapusData(${index})">DELETE</button>
        </td>
      `;
      tabel.appendChild(row);
    });

    // Fungsi hapus
    function hapusData(index) {
      if (confirm("Yakin ingin menghapus data ini?")) {
        data.splice(index, 1);
        localStorage.setItem("dataKaryawan", JSON.stringify(data));
        location.reload();
      }
    }

    // Fungsi edit
    function editData(index) {
      localStorage.setItem("editIndex", index);
      localStorage.setItem("editItem", JSON.stringify(data[index]));
      window.location.href = "edit.html";
    }
    </script>
