Kode Program: edit.html

<!DOCTYPE html>
<html>
<head>
  <title>Edit Data Karyawan</title>
</head>
<body>
  <h2>Edit Data Karyawan</h2>
  <form onsubmit="simpanPerubahan(event)">
    <label>NIM:</label><br>
    <input type="text" id="editNpm" required><br><br>

    <label>Nama:</label><br>
    <input type="text" id="editNama" required><br><br>

    <label>NoTelp:</label>
    <input type="text" id="editNotelp" required><br><br>

    <label>Alamat:</label><br>
    <input type="text" id="editAlamat" required><br><br>

    <button type="submit">Simpan</button>
    <a href="data.html"><button type="button">Batal</button></a>
  </form>

  <script>
    // Ambil data dari localStorage
    const data = JSON.parse(localStorage.getItem("dataKaryawan")) || [];
    const index = localStorage.getItem("editIndex");
    const item = JSON.parse(localStorage.getItem("editItem"));

    // Tampilkan data di input
    document.getElementById("editNpm").value = item.npm;
    document.getElementById("editNama").value = item.nama;
    document.getElementById("editNotelp").value = item.notelp;
    document.getElementById("editAlamat").value = item.alamat;

    function simpanPerubahan(event) {
      event.preventDefault();

      // Ambil nilai baru dari input
      data[index] = {
        npm: document.getElementById("editNpm").value,
        nama: document.getElementById("editNama").value,
        notelp: document.getElementById("editNotelp").value,
        alamat: document.getElementById("editAlamat").value
      };

      // Simpan ke localStorage
      localStorage.setItem("dataMahasiswa", JSON.stringify(data));

      // Bersihkan item sementara
      localStorage.removeItem("editIndex");
      localStorage.removeItem("editItem");

      alert("Data berhasil diperbarui!");
      window.location.href = "data.html";
    }
  </script>
</body>
</html>
